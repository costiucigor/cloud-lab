Customer Email System

Send personalized HTML emails to customers from Google Sheets data.

function sendCustomerEmails() {
  var file = DriveApp.getFileById("SPREADSHEET_ID");
  var spreadsheet = SpreadsheetApp.open(file);
  var data = spreadsheet.getSheets()[0].getDataRange().getValues();
  
  var template = HtmlService.createTemplateFromFile("customer-email");
  
  for (var i = 1; i < data.length; i++) {
    template.customerName = data[i][0];
    template.paymentAmount = data[i][1];
    var email = data[i][2];
    
    var htmlBody = template.evaluate().getContent();
    GmailApp.sendEmail(email, "Payment Confirmation", "", {htmlBody: htmlBody});
  }
}

Conference Registration Form

Auto-send confirmation emails when a form is submitted.

function onFormSubmit(e) {
  var response = e.response;
  var items = response.getItemResponses();
  
  var title = "", name = "", email = "", sections = [];
  
  items.forEach(function(item) {
    var question = item.getItem().getTitle().toLowerCase();
    var answer = item.getResponse();
    
    if (question.includes("title")) title = answer;
    else if (question.includes("name")) name = answer;
    else if (question.includes("email")) email = answer;
    else if (question.includes("section")) sections = answer;
  });
  
  // Send confirmation email...
}
